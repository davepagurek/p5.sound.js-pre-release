Example:

let noise, env, cnv; let types = ['white', 'pink', 'brown']; let noiseType = 'brown';

function setup() { cnv = createCanvas(100, 100); textAlign(CENTER); cnv.mousePressed(start); noise = new Noise(noiseType); env = new Envelope(0.01, 0.1, 0.15, 0.5); noise.disconnect(); noise.connect(env); noise.start(); }

function start() { noiseType = random(types); noise.type(noiseType); env.play(); }
