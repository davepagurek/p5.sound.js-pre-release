FFT Class

Defined in: src/FFT.js:3

Analyze the frequency of sound. Returns results from the frequency spectrum or time domain (waveform).

Constructor

FFT

(
  • [fftSize]
)

Defined in src/FFT.js:3

Parameters:

  • [fftSize] Number optional

    FFT anaylsis size. Must be a power of two between 16 and 1024. Defaults to 32.

Example:

let osc;

for (let i = 0; i < spectrum.length; i++) { let x = map(i, 0, spectrum.length, 0, width);
let h = -height + map(spectrum[i], 0, 0.1, height, 0); rect(x, height, width / spectrum.length, h ) }

for (let i = 0; i < waveform.length; i++){ let x = map(i, 0, waveform.length, 0, width); let y = map( waveform[i], -1, 1, 0, height); vertex(x,y); } endShape();
